Bucles

for

for (let i = 0; i < 5; i++) {
  console.log(i);
}

while / do...while

let i = 0;
while (i < 3) {
  console.log(i);
  i++;
}

let j = 0;
do {
  console.log(j);
  j++;
} while (j < 2);
