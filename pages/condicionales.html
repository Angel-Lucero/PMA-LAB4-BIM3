Condicionales

if / else

let edad = 18;
if (edad >= 18) {
  console.log('Es mayor de edad');
} else {
  console.log('Es menor de edad');
}

switch

let dia = 2;
switch (dia) {
  case 1:
    console.log('Lunes');
    break;
  case 2:
    console.log('Martes');
    break;
  default:
    console.log('Otro día');
}
